

```C++
#include <iostream>
#include <unordered_map>
using namespace std;

const int mod = 1e9 + 7;
int main() {
    int n;
    cin >> n;
    unordered_map<int, int> primes;
    while (n--) {
        int a;
        cin >> a;
        for (int i = 2; i <= a / i; ++i) {
            while (a % i == 0) {
                a /= i;
                ++primes[i];
            }
        }
        if (a > 1) ++primes[a];
    }
    long long res = 1;
    for (auto prime: primes) {
        res = res * (prime.second + 1) % mod;   //基于算数基本定理的公式，具体看思维导图
    }
    cout << res << endl;
    return 0;
}
```
