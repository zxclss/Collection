#include <iostream>
#include <algorithm>

using namespace std;

const int N = 1e5 + 10;
struct Range{
    int l, r;
    bool operator <(const Range& t) {
        return l < t.l;
    }
}ranges[N];

int main() {
    int start, end, n;
    scanf("%d%d%d", &start, &end, &n);
    for (int i = 0; i < n; ++i) {
        int l, r;
        scanf("%d%d", &l, &r);
        ranges[i] = {l, r};
    }
    sort(ranges, ranges + n);
    
    int res = 0;
    for (int i = 0; i < n; ++i) {
        int j = i, newstart = -1e9;
      //第二个指针j负责遍历之后的序列
      //找到左端点在当前left左侧的所有区间里，能到达的最远距离
        while (j < n && ranges[j].l <= start) {
            newstart = max(newstart, ranges[j].r);
            ++j;
        }
      //如果j没有更新newstart的值，说明范围内没有满足要求的区间
      //直接break，返回“没有找到”的情况即可
        if (newstart < start) {
            res = -1;
            break;
        }
      //如果找到了，就更新状态
        ++res;
        start = newstart;
        i = j - 1;
      //如果已经覆盖了当前区间，则直接返回
        if (start >= end) break;
    }
    printf("%d", start >= end ? res : -1);
    return 0;
}
