

区间选点那题用了vector实现
这里选择结构体的实现方式

```c++
#include <iostream>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10, INF = 1e9;
struct section{
    int l, r;
    bool operator<(const section& s) const {
        return r < s.r;
    }
}sections[N];
int n;

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) {
        int l, r;
        scanf("%d%d", &l, &r);
        sections[i] = {l, r};
    }
    sort(sections, sections + n);
    int res = 0, prev = -INF;
    for (int i = 0; i < n; ++i) {
        if (sections[i].l > prev) {
            ++res;
            prev = sections[i].r;
        }
    }
    printf("%d\n", res);
    return 0;
}
```
