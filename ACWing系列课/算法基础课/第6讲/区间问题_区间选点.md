#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

typedef pair<int,int> PII;
const int INF = 1e9;
int n;

int main() {
    scanf("%d", &n);
    vector<PII> vec;
    while (n--) {
        int l, r;
        scanf("%d%d", &l, &r);
        vec.push_back({r, l});
    }
    sort(vec.begin(), vec.end());
    int res = 0, prev = -INF;
    for (auto p: vec) {
        if (p.second > prev) {   //如果区间的左端点在维护的点的右侧，则说明需要更新了
            ++res;
            prev = p.first;      //更新这个区间的右端点为新的维护的点
        }
    }
    printf("%d\n", res);
    return 0;
}
