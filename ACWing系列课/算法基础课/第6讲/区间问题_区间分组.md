#include <iostream>
#include <queue>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10;
int n;
struct Range{
    int l, r;
    bool operator <(const Range& r) {
        return l < r.l;
    }
} ranges[N];

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) {
        int l, r;
        scanf("%d%d", &l, &r);
        ranges[i] = {l, r};
    }
    sort(ranges, ranges + n);
    priority_queue<int, vector<int>, greater<int>> heap;
    for (int i = 0; i < n; ++i) {
        auto t = ranges[i];
        //如果新加的元素与此前任何一个元素都有交集，则创建新组
        if (heap.empty() || heap.top() >= t.l) heap.push(t.r);
        //否则，把它加入到右端点最小的区间的集合里（贪心算法证明）
        else {
            int top = heap.top();
            heap.pop();
            heap.push(t.r);
        }
    }
    printf("%d\n", heap.size());
    return 0;
}
