## 朴素写法

```c++
#include <iostream>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10;
int n;
int a[N];

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", &a[i]);
    sort(a, a + n);
    int res = 0;
    for (int i = 0, j = n - 1; i < j; ++i, --j) {
        res += a[j] - a[i];
    }
    printf("%d\n", res);
    return 0;
}
```
## 中位数写法

```c++
    for (int i = 0; i < n; ++i) res += abs(a[i] - a[n >> 1]);
```
