vector<int> mul(const vector<int>& v1, int a) {
    vector<int> res;
    int t = 0;
    for (int i = 0; i < v1.size(); ++i) {
        t += v1[i] * a;
        res.push_back(t % 10);
        t /= 10;
    }
    if (t) res.push_back(t);
    while (res.size() > 1 && res.back() == 0) res.pop_back();
    return res;
}
