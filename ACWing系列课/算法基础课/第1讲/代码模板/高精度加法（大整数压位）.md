### 不压位高精度加法

```C++
vector<int> sum(const vector<int>& v1, const vector<int>& v2) {
    vector<int> res;
    int t = 0; //保存进位
    for (int i = 0; i < v1.size(); ++i) {
        t += v1[i];
        if (i < v2.size()) t += v2[i];
        res.push_back(t % 10);
        t /= 10;
    }
    if (t) res.push_back(t);
    return res;
}
```

**压9位代码：**（int存储的最大值是4294967295是10位数，且最大的9位数999999999相加仍然在int的范围之内，因此最大可以压9位）

```C++
#include <iostream>
#include <vector>
using namespace std;

const int Base = 1e10;

bool cmp(const vector<int>& v1, const vector<int>& v2) {
    if (v1.size() != v2.size()) return v1.size() > v2.size();
    for (int i = v1.size() - 1; i >= 0; --i) {
        if (v1[i] != v2[i]) {
            return v1[i] > v2[i];
        }
    }
    return true;
}

vector<int> sum(const vector<int>& v1, const vector<int>& v2) {
    vector<int> res;
    int t = 0; //保存进位
    for (int i = 0; i < v1.size(); ++i) {
        t += v1[i];
        if (i < v2.size()) t += v2[i];
        res.push_back(t % Base);  //基数为1e10
        t /= Base;
    }
    if (t) res.push_back(t);
    return res;
}

int main() {
    string s1, s2;
    vector<int> v1, v2, res;
    cin >> s1 >> s2;
    for (int i = s1.size() - 1, step = 0, t = 1, sum = 0; i >= 0; --i) {
        sum += (s1[i] - '0') * t;  //按10进制把9位数从string转换成int类型
        t *= 10, ++step;
        if (step == 9 || i == 0) {  //如果位数已经满9位，或者当前已经是string的第一个字符了——首位数
            v1.push_back(sum);
            sum = step = 0, t = 1; //重制
        }
    }
    for (int i = s2.size() - 1, step = 0, t = 1, sum = 0; i >= 0; --i) {
        sum += (s2[i] - '0') * t;
        t *= 10, ++step;
        if (step == 9 || i == 0) {
            v2.push_back(sum);
            sum = step = 0, t = 1; //重制
        }
    }
    if (cmp(v1, v2)) {
        res = sum(v1, v2);
    } else {
        res = sum(v2, v1);
    }
    cout << res.back();      //第一个数要单独输出，后面的数采用格式化输出，不满9位数的前面添0
    for (int i = res.size() - 2; i >= 0; --i) printf("%09d", res[i]);
    return 0;
}
```
