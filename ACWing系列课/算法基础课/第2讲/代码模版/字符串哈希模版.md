#include <iostream>
using namespace std;
const int N = 300010, P = 13331;
unsigned long long h[N], p[N];
char str[N];

unsigned long long get(int left, int right) {
    return h[right] - h[left - 1] * p[right - left + 1];
}

int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    scanf("%s", str + 1);
    p[0] = 1;
    for (int i = 1; i <= n; ++i) {
        h[i] = h[i-1] * P + str[i];
        p[i] = p[i-1] * P;
    }
    while (m--) {
        int l1, r1, l2, r2;
        scanf("%d%d%d%d", &l1, &r1, &l2, &r2);
        if (get(l1, r1) == get(l2, r2)) puts("Yes");
        else puts("No");
    }
    return 0;
}
