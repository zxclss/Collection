//⚠️这里所有的 k 表示第k - 1次被插入的元素
//因为单链表的首元素在下标为0的地方，所以第k个操作的数被存放在k - 1的位置
#include <iostream>
using namespace std;

const int N = 1e5 + 10;
int e [N], ne [N];
int head, idx, n;

void init() {
    head = -1, idx = 0;
}

void insertHead(int x) {
    e[idx] = x;
    ne[idx] = head;
    head = idx++;
}

void deleteAfterK(int k) {
    ne[k] = ne[ne[k]];
}

void insertAfterK(int k, int x) {
    e[idx] = x;
    ne[idx] = ne[k];
    ne[k] = idx++;
}

int main() {
    init();
    cin >> n;
    while (n--) {
        char oper;
        int x, k;
        cin >> oper;
        if (oper == 'H') {
            scanf("%d", &x);
            insertHead(x);
        } else if (oper == 'D') {
            scanf("%d", &k);
            if (!k) head = ne[head];
            else deleteAfterK(k - 1);
        } else {
            scanf("%d%d", &k, &x);
            insertAfterK(k - 1, x);
        }
    }
    for (int i = head; i != -1; i = ne[i]) cout << e[i] << ' ';
    cout << endl;
    return 0;
}
