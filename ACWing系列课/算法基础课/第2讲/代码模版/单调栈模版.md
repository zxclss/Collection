

```C++
//用栈维护一个单调递增序列
//要加入元素的时候就在栈顶弹出元素
//直到栈顶元素小于要加入的元素，或栈为空时
//这样每次栈顶都维护了一个当前已遍历的元素中第一个小于当前元素的元素
#include <iostream>
using namespace std;

const int N = 1e5 + 10;
int stk[N], tt = -1;

int main() {
    int n;
    scanf("%d", &n);
    while (n--) {
        int x;
        scanf("%d", &x);
      
        while (tt >= 0 && x <= stk[tt]) --tt;
        if (tt == -1) printf("-1 ");
        else printf("%d ", stk[tt]);
        stk[++tt] = x;
    }
    return 0;
}
```
